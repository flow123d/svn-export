main ->
  init_problem, make_problem (read ini file,chack, log files)
  make_mash ->
    read_node_list
    read_element_list
    read_material_list
    read_boundary_list
    (( read_initial_list
       read_concentration_list
       read_transport_bcd_list
       read_neighbour_list
       read_source_list
    ))
    make_side_list
    make_edge_list
    make_hashes
    make_element_properties
  
  topology - set and check various cross pointers 
  make_element_geometry (volume/area/length; virtual volumes; centr. points)
  preprocess 
    solver_type 
    matrix_type ( MT_WHOLE || MT_UPPER )
    (( 
       make_ele_initials ( set pressure from epress )
       transport_init_preliminary ( set init concentrations )
    ))

  ((
    main_convert_to_pos ( just do general output )
    main_compute_mh
    ((
      main_compute_mh_steady_saturated ->
         calculation_mh
         solve_system
         ((
           output_transport_init
           output_transport_time
         ))
         postprocess
         output
      
      main_compute_mh_unsteady_saturated
         output_flow_field_init
         calculation_mh
         for(( time< stop_time ))
            calculation_unsteady
            solve_system
            postprocess
            (( output_flow_field_in_time ))
          
      main_compute_mh_density
         output_transport_init
         output_transport_time
         save
    ))
  
  ))

-----

calculation_mh ->
  edge_calculation_mh ( mtx_idcs of edges (block C); values on F block - diagonal:
     f_val= -1.0 *edg->neigh_vb->coef[ 0 ] )
  element_calculation_mh 
    rows of els in A, B blocks; 
    FOR_ELEMENTS :
      calc_material_tensors ( conductivity tensor and its inversion )
      calc_rhs ( set zero )
      calc_b_val ( set -1 in block B ) 
      dirichlet_elm ( ele_rhs -= bcd->scalar at dirichlet els. )
      calc_rhs_b
      make_block_d
      make_block_e
  
  side_calculation_mh ->
     calc_side_c_col ( for each elem set cols in C block )
     FOR_SIDES( sde ) :
        calc_side_c_row ( set row in C block = edge number )
        calc_side_c_val ( )
	calc_side_metrics; calc_side_normal; calc_side_centre
        (( calc_side_rhs_dens || calc_side_rhs    ( gravitation )  ))
   
  boundary_calculation_mh
  local_matrices_mh
  make_matrix_mh
